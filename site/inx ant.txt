/* PARTICULAS CONECTADAS */
const canvas = document.getElementById('particles');
const ctx = canvas.getContext('2d');
canvas.width = window.innerWidth;
canvas.height = window.innerHeight;
let particles = [];

for(let i=0;i<80;i++){
  particles.push({ x:Math.random()*canvas.width, y:Math.random()*canvas.height, vx:(Math.random()-0.5)*0.5, vy:(Math.random()-0.5)*0.5, r:Math.random()*2+1 });
}

function animate(){
  ctx.clearRect(0,0,canvas.width,canvas.height);
  for(let i=0;i<particles.length;i++){
    for(let j=i+1;j<particles.length;j++){
      const dx = particles[i].x - particles[j].x;
      const dy = particles[i].y - particles[j].y;
      const dist = Math.sqrt(dx*dx+dy*dy);
      if(dist<100){
        ctx.strokeStyle='rgba(56,221,248,'+(1-dist/100)+')';
        ctx.lineWidth=1;
        ctx.beginPath();
        ctx.moveTo(particles[i].x,particles[i].y);
        ctx.lineTo(particles[j].x,particles[j].y);
        ctx.stroke();
      }
    }
  }
  particles.forEach(p=>{
    p.x+=p.vx; p.y+=p.vy;
    if(p.x<0||p.x>canvas.width)p.vx*=-1;
    if(p.y<0||p.y>canvas.height)p.vy*=-1;
    ctx.beginPath();
    ctx.arc(p.x,p.y,p.r,0,Math.PI*2);
    ctx.fillStyle="#38bdf8";
    ctx.fill();
  });
  requestAnimationFrame(animate);
}
animate();

/* EDITOR */
function executar(){
  const codigo=document.getElementById("editorCode").value;
  const consoleBox=document.getElementById("console");
  consoleBox.innerHTML="";
  const logOriginal=console.log;
  console.log=function(...msg){
    let linha=document.createElement("div");
    linha.innerText=msg.join(" ");
    consoleBox.appendChild(linha);
  }
  try{ eval(codigo) } catch(e){ consoleBox.innerText="Erro: "+e; }
  console.log=logOriginal;
}
document.getElementById("editorCode").addEventListener("keydown", e=>{
  if(e.ctrlKey && e.key==="Enter") executar();
});
const editor=document.getElementById("editorCode");
const linhas=document.getElementById("linhas");
function atualizarLinhas(){
  linhas.innerHTML="";
  const total=editor.value.split("\n").length;
  for(let i=1;i<=total;i++) linhas.innerHTML+=i+"<br>";
}
editor.addEventListener("input", atualizarLinhas);
atualizarLinhas();

let indice=0,pontos=0;
function carregarPergunta(){
  const pergunta=document.getElementById("pergunta");
  const respostas=document.getElementById("respostas");
  respostas.innerHTML="";
  pergunta.innerText=perguntas[indice].p;
  perguntas[indice].r.forEach((resp,i)=>{
    let btn=document.createElement("button");
    btn.innerText=resp;
    btn.onclick=()=>responder(i);
    respostas.appendChild(btn);
  });
  document.getElementById("proxima").style.display="none";
}
function responder(i){
  const botoes=document.querySelectorAll("#respostas button");
  botoes.forEach(b=>b.disabled=true);
  if(i===perguntas[indice].c){ botoes[i].style.background="#22c55e"; botoes[i].style.boxShadow="0 0 15px #22c55e"; pontos++; }

  document.getElementById("proxima").style.display="block";
}
function proximaPergunta(){

  if(indice<perguntas.length) carregarPergunta();

    document.getElementById("proxima").style.display="none";
